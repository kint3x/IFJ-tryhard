<fun_def>      -> func ID <fun_stypes> <fun_rtypes>
<fun_stypes>   -> ( <fun_stype> )
<fun_rtypes>   -> ( <fun_rtype> )
<fun_rtypes>   -> e

<fun_rtype>    -> , <type> <fun_stype>
<fun_rtype>    -> e

<stat_list>    -> <stat> <stat_list>
<stat_list>    -> e

<stat>         -> <expr> <EOL>
<stat>         -> id := <expr> <EOL>
<stat>         -> id = <expr> <EOL>
<stat>         -> if <cond> {<EOL> <stat_list> } <EOL> 
<stat>         -> if <cond> { <stat_list> } else { <EOL> <stat_list> } <EOL>
<stat>         -> return <expr> <EOL>
<stat>         -> return <cond> <EOL>
